const PAGE_SIZE = 50;

const LS_RECENT = 'sc_hangar_recent_v1';

// ── Storage ───────────────────────────────────────────────────────────────────
function loadRecent()  { try { return JSON.parse(localStorage.getItem(LS_RECENT)||'[]'); } catch { return []; } }
function saveRecent(a) { try { localStorage.setItem(LS_RECENT, JSON.stringify(a)); } catch {} }
let recents = loadRecent();

// ── File drop/pick ────────────────────────────────────────────────────────────
const dz = document.getElementById('drop-zone');
const fi = document.getElementById('file-input');
dz.addEventListener('dragover', e => { e.preventDefault(); dz.classList.add('dragover'); });
dz.addEventListener('dragleave', () => dz.classList.remove('dragover'));
dz.addEventListener('drop', e => {
  e.preventDefault(); dz.classList.remove('dragover');
  if (e.dataTransfer.files[0]) readFile(e.dataTransfer.files[0]);
});
fi.addEventListener('change', () => { if (fi.files[0]) readFile(fi.files[0]); });

function readFile(f) {
  const r = new FileReader();
  r.onload = e => parseHangar(e.target.result, f.name);
  r.readAsText(f, 'utf-8');
}

// ── Parse ─────────────────────────────────────────────────────────────────────
function parseHangar(text, filename) {
  let raw;
  try { raw = JSON.parse(text); } catch(e) { alert('Invalid JSON file. Please export from hangar.link and try again.'); return; }
  if (!Array.isArray(raw) || !raw.length) { alert('No hangar data found in this file.'); return; }

  allItems = raw.map((d, i) => {
    // Collect all manufacturers from ships
    const manufacturers = [...new Set((d.ships||[]).map(s=>s.manufacturer).filter(Boolean))];
    const shipTitles    = (d.ships||[]).map(s=>s.title).filter(Boolean);
    const skinTitles    = (d.skins||[]).map(s=>s.title).filter(Boolean);
    const fpsTitles     = (d.fps_equipment||[]).map(s=>s.title).filter(Boolean);
    const extras        = (d.hangar_extras||[]).filter(Boolean);

    return {
      id: `item_${d.pledge_id || i}`,
      pledge_id: d.pledge_id,
      type: d.type || 'other',
      source: d.source || 'Active',
      name: d.name_clean || d.name_original || '(Unknown)',
      name_original: d.name_original || '',
      value_usd: d.value_usd || 0,
      date: d.date || '',
      insurance: d.insurance || null,
      manufacturers,
      shipTitles,
      skinTitles,
      fpsTitles,
      extras,
      upgrade_from: d.upgrade_from || null,
      upgrade_to: d.upgrade_to || null,
      contains_summary: d.contains_summary || '',
      // All searchable text pre-joined
      _searchText: [
        d.name_clean, d.name_original,
        ...shipTitles, ...skinTitles, ...fpsTitles,
        ...manufacturers, d.insurance,
        d.upgrade_from, d.upgrade_to
      ].filter(Boolean).join(' ').toLowerCase()
    };
  });

  // Show loaded file
  document.getElementById('loaded-file-name').textContent = filename;
  document.getElementById('loaded-file').classList.add('visible');

  boot();
}

// ── Boot ──────────────────────────────────────────────────────────────────────
function boot() {
  document.getElementById('drop-zone').style.display = 'none';
  document.getElementById('search-area').style.display = 'block';
  document.getElementById('stats-block').style.display = 'flex';

  const active  = allItems.filter(i => i.source === 'Active');
  const buyback = allItems.filter(i => i.source === 'Buyback');

  // Active fleet stats
  const activeVal   = active.reduce((a,b) => a + b.value_usd, 0);
  const activeShips = active.filter(i => i.shipTitles.length > 0).length;
  document.getElementById('stat-active-value').textContent = '$' + Math.round(activeVal).toLocaleString();
  document.getElementById('stat-active-ships').textContent = activeShips.toLocaleString();

  // Buyback stats
  const bbVal      = buyback.reduce((a,b) => a + b.value_usd, 0);
  const bbShips    = buyback.filter(i => i.type === 'ship').length;
  const bbPacks    = buyback.filter(i => i.type === 'pack').length;
  const bbUpgrades = buyback.filter(i => i.type === 'upgrade').length;
  const bbOther    = buyback.filter(i => i.type === 'other').length;
  document.getElementById('stat-bb-value').textContent    = '$' + Math.round(bbVal).toLocaleString();
  document.getElementById('stat-bb-ships').textContent    = bbShips.toLocaleString();
  document.getElementById('stat-bb-packs').textContent    = bbPacks.toLocaleString();
  document.getElementById('stat-bb-upgrades').textContent = bbUpgrades.toLocaleString();
  document.getElementById('stat-bb-other').textContent    = bbOther.toLocaleString();

  buildMaps();
  buildTypeSidebar();
  buildMfrSidebar();
  buildDropdowns();
  wireEvents();
  applyFilters();
  document.getElementById('search-input').focus();
}

function buildMaps() {

  allItems.forEach(item => {
    typeMap[item.type] = (typeMap[item.type]||0) + 1;
    item.manufacturers.forEach(m => { mfrMap[m] = (mfrMap[m]||0) + 1; });
  });
}

// ── Sidebar selections ────────────────────────────────────────────────────────
function sidebarTypeSelect(t) {
  activeType = t;
  activeMfr  = '';
  document.getElementById('type-filter').value = t;
  document.getElementById('mfr-filter').value  = '';
  updateSidebarActive();
  currentPage = 1; applyFilters();
}

  activeMfr  = m;
  activeType = '';
  document.getElementById('mfr-filter').value  = m;
  document.getElementById('type-filter').value = '';
  updateSidebarActive();
  currentPage = 1; applyFilters();
}

function updateSidebarActive() {
  document.querySelectorAll('.sb-type-btn').forEach(el => {
    el.classList.toggle('active', el.dataset.type === activeType);
  });

    el.classList.toggle('active', el.dataset.mfr === activeMfr);
  });
}

// ── Events ────────────────────────────────────────────────────────────────────
function wireEvents() {
  const inp  = document.getElementById('search-input');
  const clrX = document.getElementById('search-clear-btn');

  inp.addEventListener('input', () => {
    clrX.style.display = inp.value ? 'block' : 'none';
    if (!inp.value && recents.length) renderRecentDropdown(true);
    else renderRecentDropdown(false);
    currentPage = 1; applyFilters();
  });
  inp.addEventListener('keydown', e => {
    if (e.key === 'Escape') { inp.blur(); renderRecentDropdown(false); }
    if (e.key === 'Enter' && inp.value.trim()) { pushRecent(inp.value.trim()); renderRecentDropdown(false); }
  });
  clrX.addEventListener('click', () => {
    inp.value = ''; clrX.style.display = 'none'; currentPage = 1; applyFilters(); inp.focus();
  });
  document.getElementById('recent-clear-btn').addEventListener('click', () => {
    recents = []; saveRecent([]); renderRecentDropdown(false);
  });
  document.addEventListener('click', e => { if (!e.target.closest('.search-wrap')) renderRecentDropdown(false); });

  document.getElementById('type-filter').addEventListener('change', e => {
    activeType = e.target.value; updateSidebarActive(); currentPage = 1; applyFilters();
  });
  document.getElementById('mfr-filter').addEventListener('change', e => {
    activeMfr = e.target.value; updateSidebarActive(); currentPage = 1; applyFilters();
  });
  document.getElementById('source-filter').addEventListener('change', () => { currentPage = 1; applyFilters(); });

  document.addEventListener('keydown', e => {
    if (e.key === '/' && document.activeElement !== inp && document.activeElement.tagName !== 'SELECT') {
      e.preventDefault(); inp.focus(); inp.select();
    }
  });
}

// ── Recent searches ───────────────────────────────────────────────────────────
function pushRecent(term) {
  recents = [term, ...recents.filter(r=>r!==term)].slice(0, MAX_RECENT);
  saveRecent(recents);
}
function renderRecentDropdown(show) {
  const drop = document.getElementById('recent-dropdown');
  if (!show || !recents.length) { drop.style.display='none'; return; }
  document.getElementById('recent-list').innerHTML = recents.map(r =>
    `<div class="recent-item" onclick="pickRecent(${JSON.stringify(r)})"><span class="recent-icon">↺</span>${esc(r)}</div>`
  ).join('');
  drop.style.display = 'block';
}
function pickRecent(term) {
  const inp = document.getElementById('search-input');
  inp.value = term;
  document.getElementById('search-clear-btn').style.display = 'block';
  renderRecentDropdown(false);
  currentPage = 1; applyFilters();
}

// ── Filter ────────────────────────────────────────────────────────────────────
function applyFilters() {
  const q      = document.getElementById('search-input').value.toLowerCase().trim();
  const type   = document.getElementById('type-filter').value;
  const mfr    = document.getElementById('mfr-filter').value;
  const source = document.getElementById('source-filter').value;
  if (q.length > 1) pushRecent(q);
  searchTerm = q;

  filtered = allItems.filter(item => {
    if (type   && item.type !== type) return false;
    if (source && item.source !== source) return false;
    if (mfr    && !item.manufacturers.includes(mfr)) return false;
    if (!q) return true;
    return item._searchText.includes(q);
  });

  renderPage();
}

// ── Render ────────────────────────────────────────────────────────────────────
function renderPage() {
  const total      = filtered.length;
  const totalPages = Math.max(1, Math.ceil(total / PAGE_SIZE));
  currentPage      = Math.min(currentPage, totalPages);
  const start      = (currentPage - 1) * PAGE_SIZE;
  const slice      = filtered.slice(start, start + PAGE_SIZE);

  const list  = document.getElementById('results-list');
  const empty = document.getElementById('empty-state');

  document.getElementById('result-count').innerHTML =
    `Showing <span>${total.toLocaleString()}</span> pledge${total !== 1 ? 's' : ''}`;

  if (!total) {
    list.innerHTML = ''; empty.style.display = 'block';
    document.getElementById('pagination').innerHTML = ''; return;
  }
  empty.style.display = 'none';

  let html = '';
  slice.forEach((c,i) => html += renderCard(c, start+i+1));

  list.innerHTML = html;
  renderPagination(totalPages);
}

function renderCard(item, num) {
  const isActive = item.source === 'Active';

  let metaHtml = '';
  if (item.source === 'Buyback') {
    metaHtml += `<span class="meta-tag source-buyback">Buyback</span>`;
  }
  if (item.insurance) {
    metaHtml += `<span class="meta-tag insurance">${esc(item.insurance)}</span>`;
  }
  if (item.manufacturers.length) {
    item.manufacturers.forEach(m => { metaHtml += `<span class="meta-tag manufacturer">${esc(m)}</span>`; });
  }
  if (item.upgrade_from && item.upgrade_to) {
    metaHtml += `<span class="meta-tag upgrade-arrow">${esc(item.upgrade_from)} → ${esc(item.upgrade_to)}</span>`;
  }
  if (item.value_usd > 0) {
    metaHtml += `<span class="meta-tag value">$${item.value_usd.toFixed(2)}</span>`;
  }

  // Summary sub-chips (collapsed view)
  let subItems = [];
  if (item.type === 'ship' || item.type === 'pack') subItems = item.shipTitles;
  else if (item.skinTitles.length) subItems = item.skinTitles;
  else if (item.fpsTitles.length)  subItems = item.fpsTitles;
  const showSubs = subItems.length > 0 && (item.type === 'pack' || subItems.length > 1 || (subItems.length === 1 && subItems[0].toLowerCase() !== item.name.toLowerCase()));
  const subHtml = showSubs
    ? `<div class="item-sub-list">${subItems.map(s=>`<span class="sub-chip">${hl(s)}</span>`).join('')}</div>` : '';

  const dateStr  = item.date ? item.date.slice(0,7) : '';
  const fullDate = item.date || '—';

  // ── Detail panel ────────────────────────────────────────────────────────────
  let detailHtml = '<div class="item-detail"><div class="detail-grid">';

  // Pledge info section
  detailHtml += `<div class="detail-section">
    <div class="detail-section-title">Pledge Info</div>
    <div class="detail-row"><span class="detail-key">Pledge ID</span><span class="detail-val">${item.pledge_id || '—'}</span></div>
    <div class="detail-row"><span class="detail-key">Type</span><span class="detail-val accent">${item.type}</span></div>
    <div class="detail-row"><span class="detail-key">Source</span><span class="detail-val ${isActive ? 'green' : 'red'}">${item.source}</span></div>
    <div class="detail-row"><span class="detail-key">Date</span><span class="detail-val">${fullDate}</span></div>
    <div class="detail-row"><span class="detail-key">Value</span><span class="detail-val green">${item.value_usd > 0 ? '$' + item.value_usd.toFixed(2) : '—'}</span></div>
  </div>`;

  // Insurance / hangar extras
  detailHtml += `<div class="detail-section">
    <div class="detail-section-title">Insurance &amp; Extras</div>
    <div class="detail-row"><span class="detail-key">Insurance</span><span class="detail-val green">${esc(item.insurance || '—')}</span></div>`;
  if (item.extras.length) {
    detailHtml += `<div class="detail-row"><span class="detail-key">Hangar Extras</span></div>
    <div class="detail-chip-row">${item.extras.map(e=>`<span class="detail-chip extra-chip">${esc(e)}</span>`).join('')}</div>`;
  } else {
    detailHtml += `<div class="detail-row"><span class="detail-key">Hangar Extras</span><span class="detail-val">—</span></div>`;
  }
  detailHtml += `</div>`;

  // Upgrade info (if applicable)
  if (item.upgrade_from || item.upgrade_to) {
    detailHtml += `<div class="detail-section">
      <div class="detail-section-title">Upgrade Path</div>
      <div class="detail-row"><span class="detail-key">From</span><span class="detail-val purple">${esc(item.upgrade_from || '—')}</span></div>
      <div class="detail-row"><span class="detail-key">To</span><span class="detail-val accent">${esc(item.upgrade_to || '—')}</span></div>
    </div>`;
  }

  detailHtml += `</div>`; // end detail-grid

  // Ships section
  if (item.shipTitles.length) {
    detailHtml += `<div class="detail-section" style="margin-bottom:0.5rem;">
      <div class="detail-section-title">Ships (${item.shipTitles.length})</div>
      <div class="detail-chip-row">${item.shipTitles.map(s=>`<span class="detail-chip ship-chip">${esc(s)}</span>`).join('')}</div>
    </div>`;
  }

  // Skins section
  if (item.skinTitles.length) {
    detailHtml += `<div class="detail-section" style="margin-bottom:0.5rem;">
      <div class="detail-section-title">Paints &amp; Skins (${item.skinTitles.length})</div>
      <div class="detail-chip-row">${item.skinTitles.map(s=>`<span class="detail-chip skin-chip">${esc(s)}</span>`).join('')}</div>
    </div>`;
  }

  // FPS Equipment section
  if (item.fpsTitles.length) {
    detailHtml += `<div class="detail-section" style="margin-bottom:0.5rem;">
      <div class="detail-section-title">FPS Equipment (${item.fpsTitles.length})</div>
      <div class="detail-chip-row">${item.fpsTitles.map(s=>`<span class="detail-chip fps-chip">${esc(s)}</span>`).join('')}</div>
    </div>`;
  }

  detailHtml += `</div>`; // end item-detail

  return `<div class="item-card" data-type="${item.type}" data-id="${item.id}" onclick="toggleCard(this)">
    <span class="card-chevron">▶</span>
    <div class="item-index">${String(num).padStart(3,'0')}</div>
    <div class="item-body">
      <div class="item-name">${hl(item.name)}</div>
      ${metaHtml ? `<div class="item-meta">${metaHtml}</div>` : ''}
      ${subHtml}
    </div>
    <div class="item-right">
      <span class="type-badge ${item.type}">${item.type}</span>
      ${dateStr ? `<div class="item-date">${dateStr}</div>` : ''}
    </div>
    ${isActive ? `<span class="active-star" title="Active — in your hangar">★</span>` : `<span class="active-star" style="opacity:0"></span>`}
    ${detailHtml}
  </div>`;
}

function toggleCard(el) {
  el.classList.toggle('expanded');
}

// ── Pagination ────────────────────────────────────────────────────────────────
function renderPagination(total) {
  if (total <= 1) { document.getElementById('pagination').innerHTML = ''; return; }
  let h = `<button class="page-btn" onclick="goPage(${currentPage-1})" ${currentPage===1?'disabled':''}>← Prev</button>`;
  const range = pgRange(currentPage, total);
  let prev = null;
  range.forEach(p => {
    if (prev !== null && p-prev > 1) h += `<span class="page-ellipsis">…</span>`;
    h += `<button class="page-btn${p===currentPage?' active':''}" onclick="goPage(${p})">${p}</button>`;
    prev = p;
  });
  h += `<button class="page-btn" onclick="goPage(${currentPage+1})" ${currentPage===total?'disabled':''}>Next →</button>`;
  document.getElementById('pagination').innerHTML = h;
}
function pgRange(cur, total) {
  const s = new Set([1, total]);
  for (let i=Math.max(2,cur-2); i<=Math.min(total-1,cur+2); i++) s.add(i);
  return [...s].sort((a,b)=>a-b);
}
function goPage(p) { currentPage = p; renderPage(); document.getElementById('results-list-wrap').scrollTo({top:0,behavior:'smooth'}); }

// ── Helpers ───────────────────────────────────────────────────────────────────
function hl(text) {
  if (!searchTerm) return esc(text);
  const re = new RegExp('('+searchTerm.replace(/[.*+?^${}()|[\]\\]/g,'\\$&')+')','gi');
  return esc(text).replace(re,'<mark>$1</mark>');
}
function esc(s) {
  return String(s).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
}
